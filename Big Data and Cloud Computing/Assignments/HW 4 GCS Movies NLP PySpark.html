Build Models: using a for loop, build a logistic regression model that predicts a label of 0 or 1 for a certain tag and repeat this for all tags. Do this for both vectorization methods for comparison.¶

# models for HTF vectorization

res_df = [] # store metric results for each model in this
tags = tags2_pd['tag'].tolist() # list of top 10 tags

evaluator_f1 = MulticlassClassificationEvaluator(metricName='f1')
evaluator_accuracy = MulticlassClassificationEvaluator(metricName='accuracy')


# Loop through each tag and run a logistic regression model on the train, val, and test dfs 
for tag in tags:
    # initialize model
    lgr = LogisticRegression(maxIter = 20, featuresCol = 'features', labelCol = tag)
    
    # train model
    lgrm = lgr.fit(htf_train_df)
    
    # run model on val and test dfs
    val_preds = lgrm.transform(htf_val_df)
    test_preds = lgrm.transform(htf_test_df)
    
    # evaluate model
    val_f1 = evaluator_f1.evaluate(val_preds, {evaluator_f1.labelCol: tag, 
                                               evaluator_f1.predictionCol: 'prediction'})
    val_accuracy = evaluator_accuracy.evaluate(val_preds, {evaluator_accuracy.labelCol: tag, 
                                                           evaluator_accuracy.predictionCol: 'prediction'})
    test_f1 = evaluator_f1.evaluate(test_preds, {evaluator_f1.labelCol: tag, 
                                                 evaluator_f1.predictionCol: 'prediction'})
    test_accuracy = evaluator_accuracy.evaluate(test_preds, {evaluator_accuracy.labelCol: tag, 
                                                             evaluator_accuracy.predictionCol: 'prediction'})
    
    # add results
    res_df.append(Row(tag = tag, val_f1 = val_f1, val_accuracy = val_accuracy, test_f1 = test_f1, 
                          test_accuracy = test_accuracy))
